85  auto& wdata = data.meta();
86  const int npts = wdata.num_pts;
87  const auto& dx = wdata.dx;
88  const auto& chord = wdata.chord;
89  const auto& aflookup = airfoil_lookup<ActTrait>(data);
90 
91  amrex::Real total_lift = 0.0;
92  amrex::Real total_drag = 0.0;
93  for (int ip = 0; ip < npts; ++ip) {
94  const auto& tmat = grid.orientation[ip];
95  // Effective velocity at the wing control point in local frame
96  auto wvel = tmat & grid.vel[ip];
97  // Set spanwise component to zero to get a pure 2D velocity
98  wvel.y() = 0.0;
99 
100  const auto vmag = vs::mag(wvel);
101  const auto aoa = std::atan2(wvel.z(), wvel.x());
102 
103  // Make up some Cl, Cd values
104  amrex::Real cl, cd;
105  aflookup(aoa, cl, cd);
106 
107  // Assume unit chord
108  const auto qval = 0.5 * vmag * vmag * chord[ip] * dx[ip];
109  const auto lift = qval * cl;
110  const auto drag = qval * cd;
111  // Determine unit vector parallel and perpendicular to velocity
112  // vector
113  const auto drag_dir = wvel.unit() & tmat;
114  const auto lift_dir = drag_dir ^ tmat.y();
115 
116  // Compute force on fluid from this section of wing
117  grid.force[ip] = -(lift_dir * lift + drag * drag_dir);
118 
119  // Assign values for output
120  wdata.vel_rel[ip] = wvel;
121  wdata.aoa[ip] = amr_wind::utils::degrees(aoa);
122  wdata.cl[ip] = cl;
123  wdata.cd[ip] = cd;
124 
125  total_lift += lift;
126  total_drag += drag;
127  }
